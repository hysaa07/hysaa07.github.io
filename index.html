Gerador de Paleta de Cores

Neste projeto, desenvolvi um Gerador de Paleta de Cores utilizando HTML, CSS e JavaScript. A aplicação permite ao usuário gerar uma paleta com cinco cores aleatórias em formato hexadecimal. Ao clicar em uma cor, o código hexadecimal correspondente é copiado para a área de transferência, facilitando sua utilização em outros projetos. Este projeto demonstra minhas habilidades no desenvolvimento de interfaces web interativas e responsivas, integrando funcionalidades práticas com um design moderno.

JS

        <script>
            const generateBtn = document.getElementById("generateBtn");
            const palette = document.getElementById("palette");
    
            // Função para gerar um código hexadecimal aleatório
            function getRandomColor() {
                return "#" + Math.floor(Math.random() * 16777215).toString(16).padStart(6, "0");
            }
    
            // Função para gerar uma paleta de 5 cores
            function generatePalette() {
                palette.innerHTML = ""; // Limpa as cores anteriores
                for (let i = 0; i < 5; i++) {
                    const color = getRandomColor();
                    const colorBox = document.createElement("div");
                    colorBox.classList.add("color-box");
                    colorBox.style.backgroundColor = color;
                    colorBox.textContent = color;
    
                    // Permitir cópia do código hexadecimal ao clicar
                    colorBox.addEventListener("click", () => {
                        navigator.clipboard.writeText(color);
                        alert(Copiado: ${color});
                    });
    
                    palette.appendChild(colorBox);
                }
            }
    
            // Gerar uma paleta ao carregar a página
            window.onload = generatePalette;
            generateBtn.addEventListener("click", generatePalette);
        </script>

  <script>
    const buttons = document.querySelectorAll(".game button");
    const currentPlayerText = document.querySelector(".currentPlayer");
    let currentPlayer = "X";

    const winningCombinations = [
      [0, 1, 2], 
      [3, 4, 5], 
      [6, 7, 8], 
      [0, 3, 6], 
      [1, 4, 7], 
      [2, 5, 8], 
      [0, 4, 8], 
      [2, 4, 6], 
    ];

    function checkWinner() {
      for (const combination of winningCombinations) {
        const [a, b, c] = combination;
        if (
          buttons[a].textContent === currentPlayer &&
          buttons[b].textContent === currentPlayer &&
          buttons[c].textContent === currentPlayer
        ) {
          return true;
        }
      }
      return false;
    }

    function checkDraw() {
      return [...buttons].every(button => button.textContent !== "");
    }

    function resetGame() {
      buttons.forEach(button => {
        button.textContent = "";
        button.disabled = false;
      });
      currentPlayer = "X";
      currentPlayerText.textContent = Vez do jogador: ${currentPlayer};
    }

    buttons.forEach((button, index) => {
      button.addEventListener("click", () => {
        button.textContent = currentPlayer;
        button.disabled = true;

        if (checkWinner()) {
          alert(Jogador ${currentPlayer} venceu!);
          resetGame();
          return;
        }

        if (checkDraw()) {
          alert("Empate!");
          resetGame();
          return;
        }

        currentPlayer = currentPlayer === "X" ? "O" : "X";
        currentPlayerText.textContent = Vez do jogador: ${currentPlayer};
      });
    });
  </script>
